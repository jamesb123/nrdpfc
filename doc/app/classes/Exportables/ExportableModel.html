141:   def exportable_join(name, *other_tables)
142:     other_tables = other_tables.flatten
143:     criteria = []
144:     r = exportable_reflections[name]
145:     raise ArgumentError, "Association #{name} doesn't exist on #{self.to_s}" if r.nil?
146:     remote_class = r.class_name.constantize
147:     criteria << remote_class.exportable_join_criteria(self, r)
148:     
149:     # check for any other joins that this table can link to - we want to make our queries as tight as possible
150:     other_relections = remote_class.exportable_reflections.values
151:     other_tables.each do |table|
152:       table = table.to_s
153:       reflection = other_relections.find{ |r| r.class_name.constantize.table_name == table && r.options[:conditions].nil? }
154:       next if reflection.nil?
155:       criteria << r.class_name.constantize.exportable_join_criteria(remote_class, reflection)
156:     end
157:     QueryPiece.new :join => "LEFT JOIN #{remote_class.exportable_table_name} ON (#{criteria.uniq * ' AND '})"
158:   end
